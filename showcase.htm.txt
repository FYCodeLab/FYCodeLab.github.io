    <script>
        const searchInput = document.getElementById('searchInput');
        const projectsGrid = document.getElementById('projectsGrid');
        const projectBoxes = document.querySelectorAll('.project-box');
        const resultsInfo = document.getElementById('resultsInfo');
        const projectCount = document.getElementById('projectCount');
        const noResults = document.getElementById('noResults');

        function updateProjectCount(count) {
            projectCount.textContent = count;
            resultsInfo.innerHTML = count === 0 ? 
                'No projects match your search' : 
                `Showing ${count} of ${projectBoxes.length} projects`;
        }

        function searchProjects() {
            const searchTerm = searchInput.value.toLowerCase().trim();
            let visibleCount = 0;

            projectBoxes.forEach(box => {
                const keywords = box.getAttribute('data-keywords').toLowerCase();
                const title = box.querySelector('.project-title').textContent.toLowerCase();
                const description = box.querySelector('.project-description').textContent.toLowerCase();
                const techTags = Array.from(box.querySelectorAll('.tech-tag')).map(tag => tag.textContent.toLowerCase()).join(' ');
                
                const searchableText = `${keywords} ${title} ${description} ${techTags}`;
                
                if (searchTerm === '' || searchableText.includes(searchTerm)) {
                    box.classList.remove('hidden');
                    visibleCount++;
                } else {
                    box.classList.add('hidden');
                }
            });

            updateProjectCount(visibleCount);
            
            if (visibleCount === 0 && searchTerm !== '') {
                noResults.style.display = 'block';
                projectsGrid.style.display = 'none';
            } else {
                noResults.style.display = 'none';
                projectsGrid.style.display = 'grid';
            }
        }

        function clearSearch() {
            searchInput.value = '';
            searchProjects();
        }

        searchInput.addEventListener('input', searchProjects);
        searchInput.addEventListener('keyup', function(event) {
            if (event.key === 'Escape') {
                clearSearch();
            }
        });

        // Initialize
        updateProjectCount(projectBoxes.length);
    </script>
